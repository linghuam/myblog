            let pair = [];
            let p1 = labels.shift();
            pair.push(p1);
            for (let i = 0; i < labels.length; i ++) {

                    pair.push(labels[i]);
                    labels.splice(i, 1);
                    break;
                }
            }
            eventPairs.push(pair);
        }

        while (pointer < eventPairs.length-1) {

                if (isSamePoint(eventPairs[pointer+1], eventPairs[pointer])) {
                   eventPairs.splice(pointer+1, 1);
                } else {
